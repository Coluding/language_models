319    def optuna_objective(self, trial):
320
321        """
322        Defines the objective function for Optuna hyperparameter optimization.
323
324        Parameters
325        ----------
326        trial : optuna.Trial
327            A single trial object for Optuna optimization.
328
329        Returns
330        -------
331        float
332            The cross-validation score for the current trial's hyperparameters.
333        
334        Raises
335        ------
336        ValueError
337            If the data type of a hyperparameter is not one of float, int, or bool.
338        """
339
340        hyperparameters = [self.n_epochs, self.learning_rate, self.classifier_dropout, self.warmup_step_fraction, self.use_gradient_clipping]
341        hyperparameter_dictionary = {}
342        for hyperparameter in hyperparameters:
343            if trial.number == 0:
344                hyperparameter_dictionary[hyperparameter.name] = hyperparameter.default
345            else:
346                if hyperparameter.dtype == float:
347                    hyperparameter_dictionary[hyperparameter.name] = trial.suggest_float(hyperparameter.name, hyperparameter.low, hyperparameter.high)
348                elif hyperparameter.dtype == bool:
349                    hyperparameter_dictionary[hyperparameter.name] = trial.suggest_categorical(hyperparameter.name, [True, False])
350                elif hyperparameter.dtype == int:
351                    hyperparameter_dictionary[hyperparameter.name] = trial.suggest_int(hyperparameter.name, hyperparameter.low, hyperparameter.high)
352                else:
353                    raise ValueError("Data type of the hyperparameters must be one of float, int or bool")
354        
355        cross_validation_score = self.cross_validate(
356            n_folds = 5,
357            training_data = self.dataset.training_data,
358            training_batch_size = 32,
359            validation_batch_size = 32,
360            n_epochs = hyperparameter_dictionary["n_epochs"],
361            learning_rate = hyperparameter_dictionary["learning_rate"],
362            classifier_dropout = hyperparameter_dictionary["classifier_dropout"],
363            warmup_step_fraction = hyperparameter_dictionary["warmup_step_fraction"],
364            use_gradient_clipping = hyperparameter_dictionary["use_gradient_clipping"]
365        )
366
367        return cross_validation_score

410        """
411
412        if not os.path.exists(self.save_path):

426            training_split = DataLoader(training_split, batch_size = training_batch_size, shuffle = False)
427            validation_split = DataLoader(validation_split, batch_size = validation_batch_size, shuffle = False)
428
429            split_score = self.train(training_split, validation_split, n_epochs, learning_rate, classifier_dropout, warmup_step_fraction, use_gradient_clipping, os.path.join(self.save_path, f"current_split_model"))
430            self.logger.info(f"Split {split + 1} is finished, the score is: {split_score:.4f}")
431            self.logger.info("-" * 100)
432            split_scores.append(split_score)
433
434        return np.mean(split_scores)

436    def train(self, training_data, validation_data, n_epochs, learning_rate, classifier_dropout, warmup_step_fraction, use_gradient_clipping, save_best_model_path):
437        
438        """
439        Trains the model on the training data and validates it on the validation data, with early stopping and checkpointing.
440
441        Parameters
442        ----------
443        training_data : DataLoader
444            The training data loader.
445        validation_data : DataLoader
446            The validation data loader.
447        n_epochs : int
448            The number of epochs to train for.
449        learning_rate : float
450            The learning rate for the optimizer.
451        classifier_dropout : float
452            The dropout rate for the classifier layer.
453        warmup_step_fraction : float
454            The fraction of steps for learning rate warm-up.
455        use_gradient_clipping : bool
456            Whether to apply gradient clipping.
457        save_best_model_path : str
458            The path to save the best model during training.
459
460        Returns
461        -------
462        float
463            The best validation score achieved during training.
464        """
465
466        # Early stopping instance with specified save path
467        early_stopping = EarlyStopping(patience = 2, verbose = True, mode = 'min', save_path = save_best_model_path)
468
469        model = self._load_electra_model(classifier_dropout)
470        model.to(self.device)
471
472        n_warmup = int(n_epochs * len(training_data) * warmup_step_fraction)
473        optimizer = torch.optim.AdamW(model.parameters(), lr = learning_rate)
474        lr_scheduler = get_linear_schedule_with_warmup(optimizer, num_training_steps = n_epochs * len(training_data), num_warmup_steps = n_warmup)
475
476        iteration = 0
477        for epoch in range(n_epochs):
478            self.logger.info(f"Starting epoch {epoch + 1}")
479            training_predictions, training_labels = [], []
480            training_loss = 0
481            for batch in training_data:
482                inputs, attention_mask, labels = batch["input_ids"].to(self.device), batch["attention_mask"].to(self.device), batch["label"].to(self.device)
483                model_output = model(input_ids = inputs, attention_mask = attention_mask, labels = labels)
484                loss = model_output.loss
485                training_loss += loss.item()
486                optimizer.zero_grad()
487                loss.backward()
488                if use_gradient_clipping:
489                    torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm = 1.0)
490                optimizer.step()
491                lr_scheduler.step()
492
493                if iteration % 20 == 0:
494                    logging.info(f"Current training batch loss: {loss.item():.4f} in epoch {epoch + 1}")
495
496                batch_predictions = model_output.logits.argmax(dim = 1)
497                training_predictions.append(batch_predictions)
498                training_labels.append(labels)
499                iteration += 1
500
501            training_loss /= len(training_data)
502            self.logger.info(f"Epoch finished, average loss over training batches: {training_loss:.4f}")
503            training_predictions = torch.cat(training_predictions, dim = 0)
504            training_labels = torch.cat(training_labels, dim = 0)
505
506            self.logger.info("-"*100)
507            self.logger.info("Training metrics:")
508            self.logger.info("-"*100)
509            if self.device.type == "cuda":
510                self._determine_scores(training_labels.cpu().numpy(), training_predictions.cpu().numpy())
511            else:
512                self._determine_scores(training_labels.numpy(), training_predictions.numpy())
513            
514            validation_predictions, validation_labels = [], []
515            validation_loss = 0
516            with torch.no_grad():
517                for batch in validation_data:
518                    inputs, attention_mask, labels = batch["input_ids"].to(self.device), batch["attention_mask"].to(self.device), batch["label"].to(self.device)
519                    model_output = model(input_ids = inputs, attention_mask = attention_mask, labels = labels)
520                    validation_loss += model_output.loss.item()
521                    batch_predictions = model_output.logits.argmax(dim = 1)
522                    validation_predictions.append(batch_predictions)
523                    validation_labels.append(labels)
524                
525            validation_loss /= len(validation_data)
526            self.logger.info(f"Average loss over validation batches: {validation_loss:.4f}")
527            validation_predictions = torch.cat(validation_predictions, dim = 0)
528            validation_labels = torch.cat(validation_labels, dim = 0)
529
530            self.logger.info("-"*100)
531            self.logger.info("Validation metrics:")
532            self.logger.info("-"*100)
533            if self.device.type == "cuda":
534                validation_scores = self._determine_scores(validation_labels.cpu().numpy(), validation_predictions.cpu().numpy())
535            else:
536                validation_scores = self._determine_scores(validation_labels.numpy(), validation_predictions.numpy())
537            max_score = validation_scores['max_score']
538
539            early_stopping(validation_loss, model)
540
541            if early_stopping.early_stop or (epoch == n_epochs - 1) and (early_stopping.best_score > -validation_loss):
542                if early_stopping.early_stop:
543                    self.logger.info("Early stopping, loading best model from before and determine score...")
544                else:
545                    self.logger.info("Last epoch reached, validation loss was better before, loading best model during training.")
546                model = self._load_electra_model(classifier_dropout, os.path.join(save_best_model_path, "checkpoint.pth"))
547                model.to(self.device)
548
549                validation_predictions, validation_labels = [], []
550                validation_loss = 0
551                with torch.no_grad():
552                    for batch in validation_data:
553                        inputs, attention_mask, labels = batch["input_ids"].to(self.device), batch["attention_mask"].to(self.device), batch["label"].to(self.device)
554                        model_output = model(input_ids = inputs, attention_mask = attention_mask, labels = labels)
555                        validation_loss += model_output.loss.item()
556                        batch_predictions = model_output.logits.argmax(dim = 1)
557                        validation_predictions.append(batch_predictions)
558                        validation_labels.append(labels)
559                validation_loss /= len(validation_data)
560                validation_predictions = torch.cat(validation_predictions, dim = 0)
561                validation_labels = torch.cat(validation_labels, dim = 0)
562
563                self.logger.info("-"*100)
564                self.logger.info("Validation metrics after reloading the model before ending this training:")
565                self.logger.info("-"*100)
566
567                if self.device.type == "cuda":
568                    validation_scores = self._determine_scores(validation_labels.cpu().numpy(), validation_predictions.cpu().numpy())
569                else:
570                    validation_scores = self._determine_scores(validation_labels.numpy(), validation_predictions.numpy())
571                max_score = validation_scores['max_score']
572                self.logger.info("Determined score from best model, ending training.")
573
574                if early_stopping.early_stop:
575                    break
576
577        return max_score

579    def final_evaluation(self, fintuned_model_path):
580        
581        """
582        Evaluates the final trained model on both the training and test datasets, returning performance metrics.
583
584        Parameters
585        ----------
586        finetuned_model_path : str
587            The path to the fine-tuned model for evaluation.
588

590        -------
591        Tuple[Dict[str, Any], Dict[str, Any]]
592            The training and test performance metrics.
593        """
594
595        model = ElectraForSequenceClassification.from_pretrained(fintuned_model_path)
